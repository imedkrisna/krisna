{{- /* Table shortcode: reads a CSV and renders as HTML table */ -}} {{- $path := .Get "path" -}} {{- $header := .Get "header" | default "true" -}} {{- $caption := .Get "caption" -}} {{- $file := .Page.Resources.GetMatch $path -}} {{- if $file -}} {{- $data := $file.Content | transform.Unmarshal (dict "delimiter" ",") -}} {{- if $caption }}{{ end -}} {{- range $i, $row := $data -}} {{- if and (eq $i 0) (eq $header "true") }} {{ range $row }}{{ end }} {{- else }} {{ range $row }}{{ end }} {{- end -}} {{- end }}
{{ $caption }}
{{ . }}
